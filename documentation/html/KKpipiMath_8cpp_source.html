21  std::vector<double> RectCoordinates(const Event &event) {
22  std::vector<double> X(5);
23  std::vector<TLorentzVector> P = ConvertEventTo4Vectors(event.GetEventVector());
24  // Use invariant mass of K+pi+ and K-pi- as variables
25  double mplus = event.GetInvMass2(0, 2);
26  double mminus = event.GetInvMass2(1, 3);
27  double mmin = std::min(mplus, mminus) - KKpipi_Constants::MASS_K - KKpipi_Constants::MASS_PI;
28  // Expand triangle in phase space into rectangle
29  X[0] = mplus + mmin;
30  X[1] = mminus + mmin;
31  // 4-vector of D meson in the rest frame
32  TLorentzVector Ptot(0.0, 0.0, 0.0, KKpipi_Constants::MASS_D);
33  // Get boost beta
34  TVector3 beta = (P[0] + P[2]).BoostVector();
35  TLorentzVector PtotTemp = Ptot, PKTemp = P[0];
36  // Boost to rest frame of K+pi+ system
37  PtotTemp.Boost(-beta);
38  PKTemp.Boost(-beta);
39  // Find angle between D meson and K+
40  X[2] = PtotTemp.Vect().Unit().Dot(PKTemp.Vect().Unit());
41  beta = (P[1] + P[3]).BoostVector();
42  PtotTemp = Ptot;
43  PKTemp = P[1];
44  // Boost to rest frame of K-pi- system
45  PtotTemp.Boost(-beta);
46  PKTemp.Boost(-beta);
47  // Find angle between D meson and K-
48  X[3] = PtotTemp.Vect().Unit().Dot(PKTemp.Vect().Unit());
49  // Find cosine and phi of the angle between decay planes of K+pi+ and K-pi-
50  double sphi = P[0].Vect().Cross(P[2].Vect()).Unit().Cross(P[1].Vect().Cross(P[3].Vect()).Unit()).Dot((P[1] + P[3]).Vect().Unit());
51  double cphi = P[0].Vect().Cross(P[2].Vect()).Unit().Dot(P[1].Vect().Cross(P[3].Vect()).Unit());
52  X[4] = TMath::ATan2(sphi, cphi);
53  return X;
54  }

56  std::vector<double> ConvertXToMomenta(const std::vector<double> &X) {
57  // Find the invariant masses of K+pi+ and K-pi-
58  double alpha = (std::min(X[0], X[1]) - KKpipi_Constants::MASS_K - KKpipi_Constants::MASS_PI)/2.0;
59  double mplus = X[0] - alpha;
60  double mminus = X[1] - alpha;
61  // Energy of K+pi+ system in the D meson rest frame
62  double Eplus = (KKpipi_Constants::MASS_D*KKpipi_Constants::MASS_D + mplus*mplus - mminus*mminus)/(2*KKpipi_Constants::MASS_D);
63  // Speed of D meson in the K+pi+ rest frame
64  double betap = TMath::Sqrt(Eplus*Eplus - mplus*mplus)/Eplus;
65  // Energy of K+ in the K+pi+ rest frame
66  double EKp = (mplus*mplus + KKpipi_Constants::MASS_K*KKpipi_Constants::MASS_K - KKpipi_Constants::MASS_PI*KKpipi_Constants::MASS_PI)/(2*mplus);
67  // Momentum of K+ in the K+pi+ rest frame
68  double pKp = TMath::Sqrt(EKp*EKp - KKpipi_Constants::MASS_K*KKpipi_Constants::MASS_K);
69  // Energy of pi+ in the K+pi+ rest frame
70  double Epip = (mplus*mplus - KKpipi_Constants::MASS_K*KKpipi_Constants::MASS_K + KKpipi_Constants::MASS_PI*KKpipi_Constants::MASS_PI)/(2*mplus);
71  // Put into four-vectors, with K+ at an angle theta+ relative to the D meson, and choose them in the xz plane
72  TLorentzVector P_Kp(pKp*TMath::Sqrt(1 - X[2]*X[2]), 0.0, -pKp*X[2], EKp);
73  TLorentzVector P_pip(-pKp*TMath::Sqrt(1 - X[2]*X[2]), 0.0, pKp*X[2], Epip);
74  // Energy of K-pi- system in the D meson rest frame
75  double Eminus = (KKpipi_Constants::MASS_D*KKpipi_Constants::MASS_D - mplus*mplus + mminus*mminus)/(2*KKpipi_Constants::MASS_D);
76  // Speed of D meson in the K-pi- rest frame
77  double betam = TMath::Sqrt(Eminus*Eminus - mminus*mminus)/Eminus;
78  // Energy of K- in the K-pi- rest frame
79  double EKm = (mminus*mminus + KKpipi_Constants::MASS_K*KKpipi_Constants::MASS_K - KKpipi_Constants::MASS_PI*KKpipi_Constants::MASS_PI)/(2*mminus);
80  // Momentum of K- in the K-pi- rest frame
81  double pKm = TMath::Sqrt(EKm*EKm - KKpipi_Constants::MASS_K*KKpipi_Constants::MASS_K);
82  // Energy of pi- in the K-pi- rest frame
83  double Epim = (mminus*mminus - KKpipi_Constants::MASS_K*KKpipi_Constants::MASS_K + KKpipi_Constants::MASS_PI*KKpipi_Constants::MASS_PI)/(2*mminus);
84  // Put into four-vector, with K- at an angle theta- relative to the D meson, and put them initially in the xz plane
85  TLorentzVector P_Km(-pKm*TMath::Sqrt(1 - X[3]*X[3]), 0.0, pKm*X[3], EKm);
86  TLorentzVector P_pim(pKm*TMath::Sqrt(1 - X[3]*X[3]), 0.0, -pKm*X[3], Epim);
87  // Boost K+ and pi+ to the D meson rest frame
88  P_Kp.Boost(0.0, 0.0, betap);
89  P_pip.Boost(0.0, 0.0, betap);
90  // Rotate the K- and pi- by phi because their decay planes are not parallel
91  P_Km.RotateZ(-X[4]);
92  P_pim.RotateZ(-X[4]);
93  // Boost I- and pi- to the D meson rest frame
94  P_Km.Boost(0.0, 0.0, -betam);
95  P_pim.Boost(0.0, 0.0, -betam);
96  // Put everything together and return it
97  std::vector<double> event = {P_Kp[0], P_Kp[1], P_Kp[2], P_Kp[3], P_Km[0], P_Km[1], P_Km[2], P_Km[3], P_pip[0], P_pip[1], P_pip[2], P_pip[3], P_pim[0], P_pim[1], P_pim[2], P_pim[3]};
98  return event;
99  }
