16 void FitGamma::DoFit(Gamma &GammaParams) {
17  if(m_minimiser == nullptr) {
18  m_minimiser = new ROOT::Minuit2::Minuit2Minimizer();
19  }
20  if(m_likelihood == nullptr) {
21  m_likelihood = new XYLikelihood(m_cpparameters);
22  }
23  m_fcn = ROOT::Math::Functor(*m_likelihood, 3);
24  m_minimiser->SetFunction(m_fcn);
25  double rB, deltaB, gamma;
26  GammaParams.GetGammaParameters(rB, deltaB, gamma);
27  m_minimiser->SetVariable(0, "r_B", rB, 0.01);
28  m_minimiser->SetVariable(1, "delta_B", deltaB, 1);
29  m_minimiser->SetVariable(2, "gamma", gamma, 1);
30  m_minimiser->Minimize();
31  const double *result = m_minimiser->X();
32  GammaParams = Gamma(result[0], result[1], result[2]);
33  double CovMatrix[16];
34  m_minimiser->GetCovMatrix(CovMatrix);
35  GammaParams.SetCov(CovMatrix);
36 }
