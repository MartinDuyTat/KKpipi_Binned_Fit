44 void SophisticatedPhaseSpace::CalculateStrongPhases(std::string BplusFilename, std::string BminusFilename, std::string MeanFilename, std::string RMSFilename) const {
45  std::vector<std::vector<double>> average(m_x3bins, std::vector<double>(m_x4bins));
46  std::vector<std::vector<double>> rms(m_x3bins, std::vector<double>(m_x4bins));
47  std::vector<std::vector<int>> numberevents(m_x3bins, std::vector<int>(m_x4bins));
48  TFile fBplus(BplusFilename.c_str(), "READ");
49  TFile fBminus(BminusFilename.c_str(), "READ");
50  TTree *treeBplus, *treeBminus;
51  fBplus.GetObject("DalitzEventList", treeBplus);
52  fBminus.GetObject("DalitzEventList", treeBminus);
53  std::vector<double> FourMomentumBplus(16), FourMomentumBminus(16);
54  for(int i = 0; i < 16; i++) {
55  std::string address = treeBplus->GetListOfBranches()[0][i]->GetName();
56  if(i%4 == 0) {
57  treeBplus->SetBranchAddress(address.c_str(), FourMomentumBplus.data() + i + 3);
58  treeBminus->SetBranchAddress(address.c_str(), FourMomentumBminus.data() + i + 3);
59  } else {
60  treeBplus->SetBranchAddress(address.c_str(), FourMomentumBplus.data() + i - 1);
61  treeBminus->SetBranchAddress(address.c_str(), FourMomentumBminus.data() + i - 1);
62  }
63  }
64  int x3bin, x4bin;
65  double BplusStrongPhase, BminusStrongPhase;
66  for(int i = 0; i < treeBplus->GetEntries(); i++) {
67  treeBplus->GetEvent(i);
68  treeBminus->GetEvent(i);
69  Event EventBplus(FourMomentumBplus), EventBminus(FourMomentumBminus);
70  BplusStrongPhase = StrongPhase(EventBplus);
71  BminusStrongPhase = StrongPhase(EventBminus);
72  x3x4WhichBinGrid(EventBplus, x3bin, x4bin);
73  numberevents[x3bin][x4bin] += 1;
74  average[x3bin][x4bin] += BplusStrongPhase;
75  rms[x3bin][x4bin] += BplusStrongPhase*BplusStrongPhase;
76  x3x4WhichBinGrid(EventBminus, x3bin, x4bin);
77  numberevents[x3bin][x4bin] += 1;
78  average[x3bin][x4bin] += BminusStrongPhase;
79  rms[x3bin][x4bin] += BminusStrongPhase*BminusStrongPhase;
80  }
81  std::ofstream AverageFile(MeanFilename), RMSFile(RMSFilename);
82  for(int i = 0; i < m_x3bins; i++) {
83  for(int j = 0; j < m_x4bins; j++) {
84  average[i][j] = average[i][j]/numberevents[i][j];
85  rms[i][j] = (rms[i][j]/numberevents[i][j]) - average[i][j]*average[i][j];
86  AverageFile << average[i][j] << " ";
87  RMSFile << rms[i][j] << " ";
88  }
89  AverageFile << std::endl;
90  RMSFile << std::endl;
91  }
92  AverageFile.close();
93  RMSFile.close();
94 }

174 int SophisticatedPhaseSpace::WhichBin(const Event &event) const {
175  std::vector<double> X = KKpipiMath::RectCoordinates(event);
176  int Region = WhichRegion(X);
177  if(Region < m_binregion) {
178  return Region;
179  }
180  int N = m_LookupBins.size();
181  // Added small correction terms to make the right edge slightly larger than pi
182  double dx1 = (RectangularPhaseSpace::GetUpperBoundary(0) + 1e-10 - RectangularPhaseSpace::GetLowerBoundary(0))/N;
183  double dx2 = (RectangularPhaseSpace::GetUpperBoundary(1) + 1e-10 - RectangularPhaseSpace::GetLowerBoundary(1))/N;
184  double dx5 = (RectangularPhaseSpace::GetUpperBoundary(4) + 1e-10 - RectangularPhaseSpace::GetLowerBoundary(4))/N;
185  int x1_bin = static_cast<int>((X[0] - RectangularPhaseSpace::GetLowerBoundary(0))/dx1);
186  int x2_bin = static_cast<int>((X[1] - RectangularPhaseSpace::GetLowerBoundary(1))/dx2);
187  int x5_bin = static_cast<int>((X[4] - RectangularPhaseSpace::GetLowerBoundary(4))/dx5);
188  int Bin = m_LookupBins[Region][x1_bin][x2_bin][x5_bin] + m_binregion;
189  return Bin;
190 }
