69 void BinList::Predict(const DDecayParameters &ddparameters, const CPParameters &cpparameters, std::vector<double> &BplusEvents, std::vector<double> &BminusEvents, int totalBplus, int totalBminus) {
70  double xplus, xminus, yplus, yminus;
71  cpparameters.GetCPParameters(xplus, xminus, yplus, yminus);
72  std::vector<double> K = ddparameters.GetK();
73  std::vector<double> Kbar = ddparameters.GetKbar();
74  std::vector<double> c = ddparameters.Getc();
75  std::vector<double> s = ddparameters.Gets();
76  BplusEvents.resize(m_bins.size());
77  BminusEvents.resize(m_bins.size());
78  double sumplus = 0, summinus = 0;
79  for(unsigned int i = 0; i < m_bins.size(); i++) {
80  BplusEvents[i] = Kbar[i] + (xplus*xplus + yplus*yplus)*K[i] + 2*TMath::Sqrt(K[i]*Kbar[i])*(xplus*c[i] - yplus*s[i]);
81  BminusEvents[i] = K[i] + (xminus*xminus + yminus*yminus)*Kbar[i] + 2*TMath::Sqrt(K[i]*Kbar[i])*(xminus*c[i] + yminus*s[i]);
82  sumplus += BplusEvents[i];
83  summinus += BminusEvents[i];
84  }
85  double normalisationBplus = totalBplus/sumplus;
86  double normalisationBminus = totalBminus/summinus;
87  std::transform(BplusEvents.begin(), BplusEvents.end(), BplusEvents.begin(), std::bind(std::multiplies<double>(), std::placeholders::_1, normalisationBplus));
88  std::transform(BminusEvents.begin(), BminusEvents.end(), BminusEvents.begin(), std::bind(std::multiplies<double>(), std::placeholders::_1, normalisationBminus));
89 }
