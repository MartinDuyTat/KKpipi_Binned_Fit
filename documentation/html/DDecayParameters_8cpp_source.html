25 DDecayParameters::DDecayParameters(PhaseSpaceParameterisation *psp, int events) {
26  // Declare necessary variables
27  Amplitude amplitude("D0toKKpipi.so", "Dbar0toKKpipi.so");
28  Double_t mass_parent = KKpipi_Constants::MASS_D;
29  Double_t mass_decay[] = {KKpipi_Constants::MASS_K, KKpipi_Constants::MASS_K, KKpipi_Constants::MASS_PI, KKpipi_Constants::MASS_PI};
30  Generator generator(mass_parent, mass_decay, 4);
31  int NumberBins = psp->NumberOfBins();
32  int counter = 0;
33  // Reset all bins that store D hadronic parameters
34  m_K = std::vector<double>(NumberBins, 0.0);
35  m_Kbar = std::vector<double>(NumberBins, 0.0);
36  m_c = std::vector<double>(NumberBins, 0.0);
37  m_s = std::vector<double>(NumberBins, 0.0);
38  // Generate events until all bins have "events" number of events, and for each event calculate the ampltitudes and phases
39  std::vector<int> EventsGenerated(NumberBins, 0);
40  while(!std::all_of(EventsGenerated.begin(), EventsGenerated.end(), [&events](int i){return i >= events;})) {
41  ++counter;
42  // Generate event
43  Event GeneratedEvent(generator.Generate());
44  // Check which bin event belongs to
45  int BinNumber = psp->WhichBin(GeneratedEvent);
46  // Increment event count for that bin
47  EventsGenerated[BinNumber] += 1;
48  // Check if bin is full
49  if(EventsGenerated[BinNumber] > events) {
50  continue;
51  }
52  // Calculate amplitude
53  std::complex<double> amplitude_d = amplitude(GeneratedEvent.GetEventVector(), +1);
54  std::complex<double> amplitude_dbar = amplitude(GeneratedEvent.GetEventVector(), -1);
55  // If amplitude is nan, event is probably on the boundary of phase space and wrongly classified as kinematically impossible, so discard event
56  if(TMath::IsNaN(std::norm(amplitude_d)) || TMath::IsNaN(std::norm(amplitude_dbar))) {
57  continue;
58  }
59  // Calculate fractional yield
60  m_K[BinNumber] += std::norm(amplitude_d);
61  m_Kbar[BinNumber] += std::norm(amplitude_dbar);
62  // Calcualte strong Phase difference
63  double phase = std::arg(amplitude_d) - std::arg(amplitude_dbar);
64  m_c[BinNumber] += TMath::Sqrt(std::norm(amplitude_d)*std::norm(amplitude_dbar))*TMath::Cos(phase);
65  m_s[BinNumber] += TMath::Sqrt(std::norm(amplitude_d)*std::norm(amplitude_dbar))*TMath::Sin(phase);
66  if(counter%events == 0) {
67  std::cout << "Generated " << counter << " events\n";
68  }
69  }
70  std::cout << "Generated " << counter << " events\n";
71  double sumK = 0, sumKbar = 0;
72  for(int i = 0; i < NumberBins; i++) {
73  // Amplitude averaged strong phase variation normalisation
74  m_c[i] /= TMath::Sqrt(m_K[i]*m_Kbar[i]);
75  m_s[i] /= TMath::Sqrt(m_K[i]*m_Kbar[i]);
76  // Account for size of phase space in each bin
77  m_K[i] *= EventsGenerated[i];
78  m_Kbar[i] *= EventsGenerated[i];
79  // Normalise fractional yields so they sum to 1
80  sumK += m_K[i];
81  sumKbar += m_Kbar[i];
82  }
83  // Divide by total to normalise fractional yields to 1
84  std::transform(m_K.begin(), m_K.end(), m_K.begin(), std::bind(std::divides<double>(), std::placeholders::_1, sumK));
85  std::transform(m_Kbar.begin(), m_Kbar.end(), m_Kbar.begin(), std::bind(std::divides<double>(), std::placeholders::_1, sumKbar));
86  std::cout << "Calculation of D hadronic decay parameters complete\n";
87 }
