20 DDecayParameters::DDecayParameters(const PhaseSpaceParameterisation &psp, const double &mass_parent, const Double_t *mass_decay, int events) {
21  Generator generator(mass_parent, mass_decay, 4);
22  BinList binlist(psp);
23  // Generate events until all bins have "events" number of events
24  // Use different events for K/Kbar and c/s
25  int counter = 0;
26  std::vector<int> v = binlist.GetEvents(+1);
27  while(!std::all_of(v.begin(), v.end(), [&events](int i){return i == events;})) {
28  std::vector<TLorentzVector> eventvectors = generator.Generate();
29  binlist.AddEvent(Event(eventvectors), +1, events);
30  v = binlist.GetEvents(+1);
31  counter++;
32  if(counter%events == 0) {
33  std::cout << "Generated " << counter << " events\n";
34  }
35  }
36  int NumberBins = binlist.NumberBins();
37  m_K = std::vector<double>(NumberBins, 0.0);
38  m_Kbar = std::vector<double>(NumberBins, 0.0);
39  m_c = std::vector<double>(NumberBins, 0.0);
40  m_s = std::vector<double>(NumberBins, 0.0);
41  Amplitude amplitude("D0toKKpipi.so", "Dbar0toKKpipi.so");
42  // Do Monte Carlo integration to find the hadronic parameters
43  for(int i = 0; i < NumberBins; i++) {
44  std::vector<Event> eventlist_d = binlist.GetBin(i).GetEvents(+1).GetEvents();
45  std::vector<Event> eventlist_dbar = binlist.GetBin(i).GetEvents(+1).GetEvents();
46  for(int j = 0; j < events; j++) {
47  // Calculate amplitude of event
48  std::complex<double> amplitude_d = amplitude(eventlist_d[j].GetEvent(), +1);
49  std::complex<double> amplitude_dbar = amplitude(eventlist_dbar[j].GetEvent(), -1);
50  // Fractional yield in bin i
51  m_K[i] += std::norm(amplitude_d);
52  m_Kbar[i] += std::norm(amplitude_dbar);
53  // Strong Phase difference
54  double phase = std::arg(amplitude_d) - std::arg(amplitude_dbar);
55  m_c[i] += TMath::Sqrt(std::norm(amplitude_d)*std::norm(amplitude_dbar))*TMath::Cos(phase);
56  m_s[i] += TMath::Sqrt(std::norm(amplitude_d)*std::norm(amplitude_dbar))*TMath::Sin(phase);
57  }
58  }
59  double sumK = 0, sumKbar = 0;
60  for(int i = 0; i < NumberBins; i++) {
61  // Normalise fractional yields so they sum to 1
62  sumK += m_K[i];
63  sumKbar += m_Kbar[i];
64  // Amplitude averaged strong phase variation normalisation
65  m_c[i] /= TMath::Sqrt(m_K[i]*m_Kbar[i]);
66  m_s[i] /= TMath::Sqrt(m_K[i]*m_Kbar[i]);
67  }
68  std::transform(m_K.begin(), m_K.end(), m_K.begin(), std::bind(std::divides<double>(), std::placeholders::_1, sumK));
69  std::transform(m_Kbar.begin(), m_Kbar.end(), m_Kbar.begin(), std::bind(std::divides<double>(), std::placeholders::_1, sumKbar));
70 }
